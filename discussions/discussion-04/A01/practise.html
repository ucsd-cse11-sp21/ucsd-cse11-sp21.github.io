  3 /*
  4     This interface defines a Volume type that can be converted to Cups, and can
  5     be added to other Volumes.
  6 */
  7 interface Volume {
  8     /*
  9         This method returns the volume of the current object in US Cups.
 10     */
 11     public double toCups();
 12 
 13     /*
 14         This method returns a new Volume which should be the sum of the 
 15         current volume, and the volume of the parameter.
 16     */
 17     public Volume add(Volume v);
 18 }
 19 
 20 /*
 21     This abstract class provides a common add() method for all
 22     classes implementing Volume by using their toCups() method
 23     to convert all Volumes to the US Cup unit, and returing the
 24     sum of all their results.
 25 */
 26 abstract class AVolume implements Volume {
 27     public Volume add (Volume v) {
 28         return new Cup(this.toCups() + v.toCups());
 29     }
 30 }
 31 
 32 /*
 33     This class represents Volumes measured in US cups.
 34 */
 35 class Cup extends AVolume {
 36     double cups;
 37 
 38     Cup(double cups) {
 39         this.cups = cups;
 40     }
 41 
 42     public double toCups() {
 43         return this.cups;
 44     }
 45 }
 46 
 47 /*
 48     This class represents Volumes measured in US tablespoons.
 49 */
 50 class Tablespoon extends AVolume {
 51     double tbsp;
 52 
 53     Tablespoon(double tbsp) {
 54         this.tbsp = tbsp;
 55     }
 56 
 57     public double toCups() {
 58         return this.tbsp / 16.0;
 59     }
 60 }

 62 /*
 63     A test class for checking our implementation of the various
 64     classes implementing Volume.
 65 */
 66 class ExampleVolumes {
 67     // Test cups
 68     Volume oneCup = new Cup(1.0);
 69     Volume twoCups = new Cup(2.0);
 70     Volume threeCups = oneCup.add(twoCups);
 71 
 72     // Test tablespoons
 73     Volume oneTablespoon = new Tablespoon(1.0);
 74     Volume twoTablespoons = new Tablespoon(2.0);
 75     
 76     /*
 77         Notice that even though we are adding two objects of type 
 78         tablespoons, here our add method implementation only returns
 79         cups, so the result of calling add here returns a Cup object!
 80     */
 81     Volume threeTablespoons = oneTablespoon.add(twoTablespoons);
 82     Volume fourTablespoons = twoTablespoons.add(twoTablespoons);
 83     
 84     // Test adding cups and tablespoons
 85     Volume fourCups = threeCups.add(sixteenTablespoons);
 86     Volume alsoFourCups = sixteenTablespoons.add(threeCups);
 87 
 88     boolean testCups(Tester t) {
 89         return t.checkExpect(1.0, oneCup.toCups()) &&
 90             t.checkExpect(2.0, twoCups.toCups()) &&
 91             t.checkExpect(3.0, threeCups.toCups());
 92     }
 93 
 94     boolean testTablespoons(Tester t) {
 95         return t.checkExpect(1.0 / 16.0, oneTablespoon.toCups()) &&
 96             t.checkExpect(2.0 / 16.0, twoTablespoons.toCups()) &&
 97             t.checkExpect(3.0 / 16.0, threeTablespoons.toCups()) &&
 98             t.checkExpect(4.0 / 16.0, fourTablespoons.toCups());
 99     }
100 
101     boolean testCoversions(Tester t) {
102         return t.checkExpect(4.0, fourCups.toCups()) &&
103             t.checkExpect(4.0, alsoFourCups.toCups()) &&
